\begin{frame}
  \frametitle{Example 1}
  What are all the books by the person who wrote "Twilight"?\\
  \hfill \\
  SELECT b1.title\\
  FROM Book b1, Book b2, Book b3\\
  WHERE b1.author = b2.author AND\\
        b3.author = b2.author AND\\
        b3.title = "Twilight";\\
\end{frame}
